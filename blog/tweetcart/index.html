Modernizing a Tweetcart

Tweetcarts are wonderful!

cls()
circfill(30,22,15,7)
::_::
flip()
srand()
rectfill(0,50,127,127,0)
for y=0,77 do
z=77/(y+1)
for i=0,z*8 do
x=(rnd(160)+t()*150/z)%160-16
w=cos(rnd()+t())*12/z
if(w>0)line(x-w,y+50,x+w,y+50,max(1,pget(x,49-y/2)))
end
end
goto _
